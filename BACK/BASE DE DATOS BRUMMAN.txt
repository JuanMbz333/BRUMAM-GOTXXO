CREATE DATA BASE BRUMMAN;
USE BRUMMAN;

CREATE TABLE ROL (
ID_ROL INT PRIMARY KEY,
NOMBRE_ROL VARCHAR (20) NOT NULL,
DESCRIPCION_ROL VARCHAR (20) NOT NULL
);

CREATE TABLE CATEGORIA (
ID_CATEGORIA INT PRIMARY KEY,
NOMBRE_CATEGORIA VARCHAR (25) NOT NULL,
DESCRIPCION_CATEGORIA VARCHAR (20) NOT NULL
);

CREATE TABLE TALLA (
ID_TALLA INT PRIMARY KEY,
TIPO_TALLA VARCHAR (5) NOT NULL,
DESCRIPCION_TALLA VARCHAR (15) NULL
);

CREATE TABLE USUARIO (
ID_USUARIO INT PRIMARY KEY,
NOMBRE_USUARIO VARCHAR (20) NOT NULL,
CORREO_USUARIO VARCHAR (30) NOT NULL,
CONTRASEÑA VARCHAR (25) NOT NULL,
TELEFONO_USUARIO VARCHAR (15) NOT NULL,
ID_ROL INT,
CREATE FOREIGN KEY (ID_ROL) REFERENCES ROL (ID_ROL)
);

CREATE TABLE PRODUCTO (
ID_PRODUCTO INT PRIMARY KEY,
DESCRIPCION_PRODUCTO VARCHAR (20) NULL,
PRECIO_PRODUCTO DECIMAL (10,2) NOT NULL,
NOMBRE_PRODUCTO VARCHAR (20) NOT NULL,
ESTADO_PRODUCTO VARCHAR (20) NOT NULL,
FECHA_CONFECCION DATE NOT NULL,
ID_CATEGORIA INT,
CREATE FOREIGN KEY (ID_CATEGORIA) REFERENCES CATEGORIA (ID_CATEGORIA)
);

CREATE TABLE PEDIDO (
ID_PEDIDO INT PRIMARY KEY,
FECHA_PEDIDO DATETIME NOT NULL,
ESTADO_PEDIDO VARCHAR (15) NOT NULL,
TOTAL_PEDIDO DECIMAL (10,2) NOT NULL
ID_USUARIO INT,
CREATE FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO (ID_USUARIO)
);

CREATE TABLE PRODUCTO TALLA (
ID_PRODUCTO_TALLA INT PRIMARY KEY,
STOCK_PRODUCTO_TALLA DECIMAL (10,2) NOT NULL,
CREATE FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTO (ID_PRODUCTO)
);

CREATE TABLE INVENTARIO (
ID_INVENTARIO INT PRIMARY KEY,
STOCK_DISPONIBLE VARCHAR (20) NOT NULL,
ID_PRODUCTO_TALLA INT,
CREATE FOREIGN KEY (ID_PRODUCTO_TALLA) REFERENCES PRODUCTO_TALLA (ID_PRODUCTO_TALLA)
);

CREATE TABLE MEDIDAS_CORPORALES (
ID_MEDIDAS_CORPORALES INT PRIMARY KEY,
ALTURA DECIMAL (5,2) NOT NULL,
CINTURA DECIMAL (5,2) NOT NULL,
CADERA DECIMAL (5,2) NOT NULL,
FECHA_REGISTRO DATE NOT NILL,
PESO DECIMAL (10,3) NOT NULL,
GENERO VARCHAR (15) NOT NULL,
ID_USUARIO INT,
CREATE FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO (ID_USUARIO)
);

CREATE TABLE MODELO_3D (
ID_MODELO_3D INT PRIMARY KEY,
MEDIDAS_MODEL DECIMAL (10,2) NOT NULL,
DESCRIPCION_MODEL VARCHAR (20) NULL,
ID_PRODUCTO INT,
CREATE FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTO (ID_PRODUCTO)
);

CREATE TABLE PAGOS (
ID_PAGOS INT PRIMARY KEY,
METODO_PAGO VARCHAR (20) NOT NULL,
FECHA_PAGO DATETIME NOT NULL,
MONTO DECIMAL (10,2) NOT NULL,
ESTADO_PAGO VARCHAR (15) NO NULL,
ID_PEDIDO INT,
CREATE FOREIGN KEY (ID_PEDIDO) REFERENCES PEDIDO (ID_PEDIDO)
);

CREATE TABLE CARRITO (
ID_CARRITO INT PRIMARY KEY,
TOTAL_CARRITO DECIMAL (10,2) NOT NULL,
ID_USUARIO INT,
CREATE FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO (ID_USUARIO)
);

CREATE TABLE ENVIO (
ID_ENVIO INT PRIMARY KEY,
DIRRECCION_ENVIO VARCHAR (10) NOT NULL,
ESTADO_ENVIO VARCHAR (15) NOT NULL, 
FECHA_ENTREGA DATETIME NOT NULL,
TRASNPORTADORA VARCHAR (20) NOT NULL,
NUMERO_GUIA VARCHAR (15) NOT NULL,
CREATE FOREIGN KEY (ID_PEDIDO) REFERENCES PEDIDO (ID_PEDIDO)
);

CREATE TABLE DEVOLUCION (
ID_DEVOLUCION INT PRIMARY KEY,
MOTIVO VARCHAR (30) NOT NULL,
ESTADO_DEVOLUCION (15) NOT NULL, 
CREATE FOREIGN KEY (ID_PEDIDO) REFERENCES PEDIDO (ID_PEDIDO)
);

CREATE TABLE NOTIFICACION ( 
ID_NOTIFICACION INT PRIMARY KEY,
MENSAJE VARCHAR (25) NOT NULL,
FECHA_NOTIFICACION DATETIME NOT NULL,
ID_USUARIO INT,
CREATE FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO (ID_USUARIO)
);

CREATE TABLE PRODUCTO_TALLA (
ID_PRODUCTO_TALLA INT PRIMARY KEY,
STOCK DECIMAL (10,2) NOT NULL,
ID_PRODUCTO INT,
ID_TALLA INT, 
CREATE FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTO (ID_PRODUCTO),
CREATE FOREIGN KEY (ID_TALLA) REFERENCES TALLA (ID_TALLA)
);

CREATE TABLE CARRITO_PRODUCTO_TALLA (
ID_CARRITO_PRODUCTO_TALLA INT PRIMARY KEY,
CANTIDAD VARCHAR (20) NOT NULL,
ID_CARRITO INT,
ID_PRODUCTO_TALLA INT,
PRIMARY KEY (ID_CARRITO , ID_PRODUCTO_TALLA),
FOREIGN KEY (ID_CARRITO) REFERENCES CARRITO (ID_CARRITO),
FOREIGN KEY (ID_PRODUCTO_TALLA) REFERENCES PRODUCTO_TALLA (ID_PRODUCTO_TALLA)
);

CREATE TABLE PEDIDO_PRODUCTO_TALLA (
ID_PEDIDO_PRODUCTO_TALLA INT PRIMARY KEY,
CANTIDAD VARCHAR (20) NOT NULL,
PRECIO_UNITARIO DECIMAL (10,2) NOT NULL
ID_PRODUCTO INT,
ID_PRODUCTO_TALLA INT,
PRIMARY KEY (ID_PRODUCTO , ID_PRODUCTO_TALLA),
FOREIGN KEY (ID_PRODUCTO) REFERENCES PRODUCTO (ID_PRODUCTO),
FOREIGN KEY (ID_PRODUCTO_TALLA) REFERENCES PRODUCTO_TALLA (ID_PRODUCTO_TALLA)
);
